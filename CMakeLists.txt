project(echo_server)

find_package(Threads REQUIRED)

    link_libraries(ws2_32 wsock32)
endif (WIN32)

if (UNIX AND NOT APPLE)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -pthread")
endif ()
